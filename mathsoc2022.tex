	\begin{frame}{測度$0$集合の和集合}
		$(Y, \mu)$をポーランド確率空間とする。
		測度の可算加法性より、測度$0$集合の可算個の和集合は再び測度$0$である。
		
		\vspace{0.5cm}
		Q. How about the union of continuum many measure $0$ sets?
		
		A. It is not necessary a measure $0$ set (e.g. singletons for all points).
		
		\vspace{0.5cm}
		Q. What if we add assumptions to the given measure $0$ sets?
		
		A. If we do it in a sense, then the union of continuum many measure $0$ sets is also a measure $0$ set!
	\end{frame}
		
	\begin{frame}{Goldsternの定理}
		\begin{block}{}
			\textbf{(full domination order)} For $x, x' \in \omega^\omega$, define a relation $x \le x'$ by $(\forall n\in\omega)(x(n) \le x'(n))$.
		\end{block}

				In 1993,  Martin Goldstern proved the following theorem.
				\begin{thmblock}{Goldstern's theorem ($\ZF+\DC$)}
					Let $(Y, \mu)$ be a Polish probability space.
					Let $A \subset \omega^\omega \times Y$ be a $\boldsig^1_1$ set.
					Assume that for each $x \in \omega^\omega$,
					\[
					A_x := \{ y \in Y : (x, y) \in A \}
					\]
					has measure $0$.
					Also, assume $(\forall x, x' \in \omega^\omega)(x \le x' \Rightarrow A_{x} \subset A_{x'})$.
					Then $\bigcup_{x \in \omega^\omega} A_x$ has also measure $0$.
				\end{thmblock}
	\end{frame}

	\begin{frame}{原理$\GP(\Gamma)$}
		\begin{defiblock}{定義}
			Let $\Gamma$ be a pointclass.
			Then \alert{$\GP(\Gamma)$} means the following statement:
			Let $(Y, \mu)$ be a Polish probability space  and $A \subset \omega^\omega \times Y$ be in $\Gamma$.
			Assume that for each $x \in \omega^\omega$, $A_x$ has $\mu$-measure $0$.
			Also suppose that $(\forall x, x' \in \omega^\omega)(x \le x' \Rightarrow A_{x} \subset A_{x'})$.
			Then $\bigcup_{x \in \omega^\omega} A_x$ has also $\mu$-measure $0$.
		\end{defiblock}
	
		Goldstern's theorem says that $\GP(\boldsig^1_1)$ holds.
		
		Note that if $\Gamma$ is a sufficiently high pointclass (that is if $\bolddelta^1_1 \subset \Gamma$), then we can assume that $(Y, \mu)$ is the Cantor space with the standard measure.
		
		%We consider $\GP(\all)$, that is Goldstern's principle for all subsets.
	\end{frame}
	\begin{frame}{主結果}
		The symbol ``$\all$" denotes the class of all subsets of Polish spaces.
		\begin{thmblock}{定理}
			$\GP(\all)$ is independent from $\ZFC$.
		\end{thmblock}
		
	\end{frame}
		
	\begin{frame}{$\neg \GP(\all)$の無矛盾性}
		\begin{columns}
			\begin{column}{0.58\textwidth}
				\begin{thmblock}{定理}
					Assume $\CH$. Then $\neg \GP(\all)$ holds.
				\end{thmblock}
				Proof. Let $\seq{x_\alpha : \alpha < \omega_1}$ be a cofinal increasing sequence in $(\omega^\omega, <^*)$. And let $\seq{y_\alpha : \alpha < \omega_1}$ be an enumeration of $2^\omega$.
				Then the set $A$ defined by the following equation witnesses $\neg \GP(\all)$:
				\[
				A_x = \{ y_\beta : \beta < \alpha_x  \},
				\]
				where $\alpha_x = \min \{\alpha : x <^* x_\alpha \}$. \qed
			\end{column}
			\begin{column}{0.48\textwidth}
				\includegraphics[width=5.5cm]{IMG_0149.jpg}
			\end{column}
		\end{columns}
	\end{frame}

	\begin{frame}{$\neg \GP(\all)$の無矛盾性}
	Refining the last proof, we get the following theorem.
	\begin{thmblock}{定理}
		Assume that at least one of the following three conditions holds:
		
		\hspace{2em}$\add(\nul) = \frakb$, $\non(\nul) = \frakb$ or $\non(\nul) = \frakd$.
		
		Then $\neg \GP(\all)$ holds.
	\end{thmblock}
	\begin{block}{}
	{\scriptsize
		$\add(\nul) := \min \{ \kappa : \text{the null ideal is not } \kappa\text{-additive} \}$
		$\non(\nul) := \min \{ \abs{A} : A \subset 2^\omega, A \text{ does not have measure } 0 \}$
		$\frakb := \min \{ \abs{F} : F \subset \omega^\omega, \neg (\exists g \in \omega^\omega)(\forall f \in F) \ f <^* g \}$
		$\frakd := \min \{ \abs{F} : F \subset \omega^\omega, (\forall g \in \omega^\omega)(\exists f \in F) \ g <^* f \}$
	}
	\end{block}
	\end{frame}

	\begin{frame}{決定性とSolovayモデル}
		\begin{thmblock}{定理}
			Assume $\ZF + \AD$. Then $\GP(\all)$ holds.
		\end{thmblock}

			Solovayモデルにおいて$\GP(\all)$が成り立つ.

	\begin{frame}{まとめ}
		\begin{itemize}
			\item Goldstern showed $\GP(\boldsig^1_1)$.
			\item We proved the following 3 theorems:
			\begin{enumerate}
				\item $\GP(\all)$ is independent from $\ZFC$.
				\item $V=L$ implies $\neg \GP(\Delta^1_2)$.
				\item Under $\ZF+\AD$, $\GP(\all)$ holds.
			\end{enumerate}
		\end{itemize}
	\end{frame}
